Create Date table 9

Use the following to create the date table:

Date Dax

Date = ADDCOLUMNS (

 

CALENDAR (Min (‘Real_Estate_Sales_2001-2019_GL'[Date Recorded]), MAX('Real_Estate_Sales_2001-2019_GL'[Date Recorded])),

 

"Datacenter", FORMAT ([Date], "YYYYMMDD”),

 

"Year", YEAR ([Date]),

 

"Month number", FORMAT ([Date], "MM”),

 

"Year Month number", FORMAT ([Date], "YYYY/MM”),

 

"YearMonthShort", FORMAT ([Date], "YYYY/mmm”),

 

"MonthNameShort", FORMAT ([Date], "mmm”),

 

"MonthNameLong", FORMAT ([Date], "mmmm”),

 

"DayOfWeekNumber", WEEKDAY ([Date]),

 

"Day of Week", FORMAT ([Date], "dddd”),

 

"DayOfWeekShort", FORMAT ([Date], "ddd”),

 

"Quarter", "Q" & FORMAT ([Date], "Q”),

 

"Year Quarter", FORMAT ([Date], "YYYY”) & "/Q" & FORMAT ([Date], "Q" )

 

)
